Binary Convert(二进制之转换)

  2021-11-25   算法   二进制 字数统计: 1231字 阅读时长: 3分

关于本人之前遇到的一个位运算问题，简单的作了下分析，如有错误，望指正。

/**
     * 二进制 逢二进一
     * 八进制 逢八进一
     * 十进制 逢十进一
     * 十六进制 逢十六进一 A(10) B(11) C(12) D(13) E(14) F(15)
     */


    /**
     * 十进制 转 二进制
     *将 十进制数 除以 2 ，取每次的余数 再将余数倒排 即为对应的二进制
     */


    /**
     * 十进制 转 八进制
     * 将 十进制数 除以 8 ，取每次的余数 再将余数倒排 即为对应的八进制
     */


    /**
     * 十进制 转 十六进制
     * 将 十进制数 除以 16 ，取每次的余数 再将余数倒排 即为对应的十六进制
     */


    /**
     * 二进制 转 十进制
     * 将二进制的每个权重值 相加 即为十进制
     * 如 1 0110-> 1*2^4+0*2^3+1*2^2+1*2^1+0*2^0
     */

    /**
     * 二进制 转 八进制
     * 将二进制从右往左 每三位（权重值） 表示一位八进制数，不足时 空位补0
     */

    /**
     * 二进制 转 十六进制
     * 将二进制从右往左 每四位（权重值） 表示一位十六进制数，不足时 空位补0
     */


    /**
     * 八进制 转 二进制
     * 将每位八进制数 用三位二进制数（必须是三位，不足时，左边补0）表示
     * 或将每位八进制数 除以2 取每次余数 再倒排 即为对应二进制数
     */

    /**
     * 八进制 转 十进制
     * 将每位八进制数 按权重值 相加 即为对应的十进制
     * 如 226  -> 2*8^2+2*8^1+6*8^0
     */

    /**
     * 八进制 转 十六进制
     * 可借助 十进制 或者 二进制 过渡
     */

    /**
     * 十六进制 转 二进制
     * 将每位十六进制数 用四位二进制数（必须是四位，不足时，左边补0）表示
     * 或将每位十六进制数 除以2 取每次余数 再倒排 即为对应二进制数
     * 96 -> 1001 0110
     */

    /**
     * 十六进制 转 十进制
     * 将每位十六进制数 按权重值 相加 即为对应的十进制
     * 如 96  -> 9*16^1+6*16^0
     */

    /**
     * 十六进制 转 八进制
     * 可借助 十进制 或者 二进制 过渡
     */

    /**
     *
     *
     * 1byte = 8bit
     * char = 2byte
     * short = 2byte
     * int = 4byte
     * long = 8byte
     *
     */


    public static void main(String[] args) {

        /**
         * byte 与 bit 结合位运算 获取 byte 中每一位bit 的表示情况
         */
        //1、接受字节流，获取每个字节中每一位的值（bit）: byte -> bite
        String str = "1234A";
//      49:  0011 0001
//        50:     0011 0010
//              0 0011 001      右移一位
//             00 0011 00       右移两位
//            000 0011 0        右移三位
//           0000 0011          右移四位
//          00000 001           右移五位
        System.out.println("bit -> byte = [" + Integer.parseInt("00110010",2) + "]");
        for(byte b:str.getBytes()){
            //取第一位的值
            System.out.println(b+" & 1 = " + (b & 1));
            //第二位
            System.out.println(b+">>1 & 1 = " + (b>>1 & 1) );
            //第三位，以此类推
            System.out.println(b+">>2 & 1 = " + (b>>2 & 1) );
            //第四位
            System.out.println(b+">>3 & 1 = " + (b>>3 & 1) );
            //第五位
            System.out.println(b+">>4 & 1 = " + (b>>4 & 1));
            //第六位
            System.out.println(b+">>5 & 1 = " + (b>>5 & 1));
            //第七位
            System.out.println(b+">>6 & 1 = " + (b>>6 & 1));
            //第八位
            System.out.println(b+">>7 & 1 = " + (b>>7 & 1));
        }


        //2、bite -> byte,未考虑负数情况，主要以左起第一位区分：0正。负数时 结果=Integer.parseInt()-256

        String[] bitArr = {"00110001","00110010"};
        for(String bit : bitArr){
            // bit -> byte
            System.out.println(bit+" -> byte = " + (byte)Integer.parseInt(bit,2));//2：说明要转换的bit是二进制，默认是十进制
            //byte -> String
            System.out.println((byte)Integer.parseInt(bit,2)+"->String = " + new String(new byte[]{(byte)Integer.parseInt(bit,2)}) );

        }

        /**
         * 位 运算
         * & | ^ ～ >> <<
         */

        //1、异或：按位异或，不同得1，同得0

        //两数交换:规则，两数异或后的结果再与其中一个数异或得到另一个数
        int a=1,b=2;
        a = a^b;
        b = a^b;
        a = a^b;
        System.out.println("a=1,b=2 -> a="+a+",b="+b);

        //一个数与另一个数异或两次结果还是这个数
        int a1=1,b1=2;
        System.out.println("a1=1,b1=2;a1^b1^b1= -> "+(a1^b1^b1));

        //2、& ：全为1得1，否则得0


        //3、| ：有1则得1

        //4、a>>n 左补n个0,缩小原来的2^n ,a = a/2^n
        int a4=16;//0001 0000
        System.out.println("a4>>2(0000 0100) = "+(a4>>2));

        //5、a<<n 右补n个0，增大为原来的 2^n ,a = a*2^n
        int a5=2;//0000 0010
        System.out.println("a5<<2(0000 1000) = "+(a5<<2));

    }
